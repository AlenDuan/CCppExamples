Clock.cpp

class Clock{
public:
    void setTime(int h,int m,int s){
        this->h = h;
        this->m = m;
        this->s = s;
    }
private:
    int h,m,s;
};

Clock.cpp

int main()
{
    Clock c;
    c.setTime(12,30,0);
    return 0;
}

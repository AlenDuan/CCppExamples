{
    return x > y ? x : y;

Stack+Vector(stl)

template <class T>
class Stack{
public:
    void push(T item);
    T pop();
    bool isEmpty();
private:
    Vector<T> data;
};

Stack+Vector

template <class T>
void Stack::push(T item){
    data.push_back(item);
}

template <class T>
T Stack::pop(){
    T value = data.back();
    data.pop_back();
    return value;
}

bool Stack::isEmpty(){
    return data.empty();
}
int main(){
    Stack<int> s1;
    s1.push(100);
    
    Stack<float> s2;
    s2.push(2.0);
}

函数指针

int add ( int a , int b ) ; //普通函数

int ( *fptr ) ( int a , int b ) ;   //定义函数指针

fptr = add ;      // 给函数指针赋值，或fptr = &add

add ( 1 , 2 ) ;

( *fptr ) ( 1 , 2 ) ; //用指向函数的指针调用函数add

fptr ( 1 , 2 ) ;      //用指向函数的指针调用函数add

static(静态)成员

class Point{
public:
    Point(){
        count++;
    }
    ~Point(){
        count--;
    }
private:
    static int count;
};
int Point::count = 0;

singleton(单例模式)

class MyClass{
public:
    static MyClass *getInstance(){
        if(instance==NULL)
            instance=new MyClass();
        return instance;
    }
private:
    static MyClass *instance;
    MyClass(){}
};
MyClass *MyClass::instance=NULL;
void main()
{
    MyClass *obj;
    obj=MyClass::getInstance();
}

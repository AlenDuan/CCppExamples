define/declare(定义/声明）: Queue类 （ Queue.h ）

class Queue{

public:
    void append ( int item ) ;
    int pop ( ) ;
    bool isEmpty ( ) ;
    bool isFull ( ) ;
private:
    int data[ 100] ;
    int head = 10 ;
    int tail = 10 ;
};

implement(实现) Queue类 ( Queue.cpp )

#include “Queue.h”

void Queue::append( int item ){
    data[ head ] = item;
    head = (head+1) % 100;
}
int Queue::pop(){
    tail = ( tail + 1 ) % 100 ;
    return data[ tail-1 ] ;
}
bool Queue::isEmpty(){
    return (head == tail);
};
bool Queue::isFull(){
    return ((head+1)%100 == tail);
};

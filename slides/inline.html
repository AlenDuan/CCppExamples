inline(内联)函数

#include <iostream>
using namespace std;
inline int max( int x , int y )
{
    return ( x>y ) ? x : y;
}
void main()
{
    int a , b ;
    cin >> a >> b ;
    cout << max( a , b ) << endl ;
}

【例3-8】Stack + Node

class Stack{
public:
    void push(int item);
    int pop();
    bool isEmpty();
private:
    Node head;
};

Stack+Node

void Stack::push(int item){
    Node * node = new Node(item);
    node->next = head.next;
    head.next = node;
}
int pop(){
    Node *first = head.next;
    int value =first->value;
    head.next = first->next;
    delete first;
    return value;
}
bool isEmpty(){
    return head.next == null;
}

【例3-9】Stack+Stack

class Stack{
public:
    void push(int item);
    int pop();
    bool isEmpty();
private:
    int data[100];
    int count = 0;
    Stack *next;
};

Stack+Stack

void Stack::push(int item){
    if (count<100){
        data[count]=item;
    }else{
        if(count == 100){
            next = new Stack();
        }
        next->push(item);
    }
    count++;
}

Stack+Stack

int pop(){
    int value;
    count--;
    if (count<100){
        value = data[count];
    }else{
        item = next->pop();
        if(count == 100){
            delete next;
        }
    }
    return value;
}
bool isEmpty(){
    return 0 == count;
}

【例3-10-1】Stack+Vector(stl)

class Stack{
public:
    void push(int item);
    int pop();
    bool isEmpty();
private:
    Vector data;
};

Stack+Vector

void Stack::push(int item){
    data.push_back(item);
}

int Stack::pop(){
    int value = data.back();
    data.pop_back();
    return value;
}
bool Stack::isEmpty(){
    return data.empty();
}
